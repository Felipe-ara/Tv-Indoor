function loadContent() { fetch('config.json') .then(response => { if (!response.ok) { throw new Error('Erro ao carregar config.json: ' + response.status); } return response.json(); }) .then(data => { if (!Array.isArray(data) || data.length === 0) { throw new Error('config.json está vazio ou não é um array válido'); } items = data; currentIndex = 0; // resetar índice showNext(); }) .catch(error => { console.error(error); const container = document.getElementById('content'); container.innerHTML = `

Erro ao carregar conteúdo:
${error.message}

`; }); }
